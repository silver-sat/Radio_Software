50int Efuse::overcurrent(bool transmit)
51{
52 //set threshold
53 float _threshold = _OC_threshold_receive;
54 if (transmit == true)
55 {
56 _threshold = _OC_threshold_transmit;
57 }
58
59 //measure to see if there's an overcurrent and report
60 int imon_reading = analogRead(_pin_5V_current);
61 float imon_voltage = imon_reading * _adc_resolution * 1000;
62 float current = (imon_voltage - _imon_intercept) / _imon_slope * 1000;
63 // if the current is greater than the threshold and the timer has expired, or if the fault line is low, then return true
64 if (((current > _threshold) && (millis()-m_repeat_timer > 500)) || !digitalRead(_pin_OC5V))
65 {
66 //m_repeat_timer is initialized when efuse.begin() is executed
67 //we have an overcurrent, so send the packet and reset the timer, don't resend until timer value is greater than 500mS
68 byte resetpacket[] = {0xC0, 0x0F, 0xC0}; // generic form of nack packet
69 #ifdef _SILVERSAT_
70 Serial0.write(resetpacket, 3);
71 #endif
72 m_repeat_timer = millis();
73 return 1;
74 }
75 else return 0;
76}
