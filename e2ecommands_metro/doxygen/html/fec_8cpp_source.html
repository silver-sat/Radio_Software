110void
111rs_encode(uint8_t *parity, const uint8_t *data, size_t len)
112{
113 memset(parity, 0, sizeof(uint8_t) * PRT_LENGTH);
114 uint8_t pad = 223 - len;
115 uint8_t feedback;
116 for (int i = 0; i < NN - PRT_LENGTH - pad; i++) {
117 feedback = index_of[data[i] ^ parity[0]];
118 if (feedback != FF) {
119 for (int j = 1; j < PRT_LENGTH; j++) {
120 parity[j] ^= alpha_to[mod255(feedback + genpoly[PRT_LENGTH - j])];
121 }
122 }
123 memmove(&parity[0], &parity[1], sizeof(uint8_t) * (PRT_LENGTH - 1));
124 if (feedback != FF) {
125 parity[PRT_LENGTH - 1] = alpha_to[mod255(feedback + genpoly[0])];
126 } else {
127 parity[PRT_LENGTH - 1] = 0;
128 }
129 }

139int
140rs_decode(uint8_t *data, size_t len)
141{
142 uint8_t pad = 0;
143 uint8_t q;
144 uint8_t tmp;
145 uint8_t num1;
146 uint8_t num2;
147 uint8_t den;
148 uint8_t temp_r;
149 uint8_t l_arr[PRT_LENGTH + 1];
150 uint8_t s[PRT_LENGTH];
151 uint8_t no_eras = 0;
152 uint8_t b[PRT_LENGTH + 1];
153 uint8_t t[PRT_LENGTH + 1];
154 uint8_t evaluator[PRT_LENGTH + 1];
155 uint8_t root[PRT_LENGTH];
156 uint8_t reg[PRT_LENGTH + 1];
157 uint8_t loc[PRT_LENGTH];
158 int16_t l_val;
159 int16_t conf_2;
160 int16_t eval_val;
161 int16_t i;
162 int16_t j;
163 int16_t conf_1;
164 int16_t k;
165 int16_t syndrome_flag;
166 int16_t corrected_bits;
167
168 syndrome_flag = 0;
169 pad = 255 - len;
170
171 if (!pad_valid(pad)) {
172 return (-1);
173 }
174
175 //Syndrome Generation
176 for (i = 0; i < PRT_LENGTH; i++) {
177 s[i] = data[0];
178 }
179 for (j = 1; j < NN - pad; j++) {
180 for (i = 0; i < PRT_LENGTH; i++) {
181 if (s[i] == 0) {
182 s[i] = data[j];
183 } else {
184 s[i] = data[j] ^ alpha_to[MODFF(index_of[s[i]] + (FCR_RS + i) * PRIM)];
185 }
186 }
187 }
188
189 //Syndrome Transformation
190 for (i = 0; i < PRT_LENGTH; i++) {
191 syndrome_flag |= s[i];
192 s[i] = index_of[s[i]];
193 }
194
195 if (syndrome_flag == 0) {
196 return 0;
197 }
198
199 memset(&l_arr[1], 0, PRT_LENGTH * sizeof(l_arr[0]));
200
201 l_arr[0] = 1;
202 for (i = 0; i < PRT_LENGTH + 1; i++) {
203 b[i] = index_of[l_arr[i]];
204 }
205
206 //Error Locator
207 conf_1 = 0;
208 conf_2 = 0;
209 while (++conf_1 <= PRT_LENGTH) {
210 temp_r = 0;
211 for (i = 0; i < conf_1; i++) {
212 if ((l_arr[i] != 0) && (s[conf_1 - i - 1] != FF)) {
213 temp_r ^= alpha_to[MODFF(index_of[l_arr[i]] + s[conf_1 - i - 1])];
214 }
215 }
216 temp_r = index_of[temp_r];
217 if (temp_r == FF) {
218 memmove(&b[1], b, PRT_LENGTH * sizeof(b[0]));
219 b[0] = FF;
220 } else {
221 t[0] = l_arr[0];
222 for (i = 0; i < PRT_LENGTH; i++) {
223 if (b[i] != FF) {
224 t[i + 1] = l_arr[i + 1] ^ alpha_to[MODFF(temp_r + b[i])];
225 } else {
226 t[i + 1] = l_arr[i + 1];
227 }
228 }
229 if (2 * conf_2 <= conf_1 + no_eras - 1) {
230 conf_2 = conf_1 + no_eras - conf_2;
231
232 for (i = 0; i <= PRT_LENGTH; i++) {
233 b[i] = (l_arr[i] == 0) ? FF : MODFF(index_of[l_arr[i]] - temp_r + NN);
234 }
235 } else {
236 memmove(&b[1], b, PRT_LENGTH * sizeof(b[0]));
237 b[0] = FF;
238 }
239 memcpy(l_arr, t, (PRT_LENGTH + 1) * sizeof(t[0]));
240 }
241 }
242
243 //Array Transformation
244 l_val = 0;
245 for (i = 0; i < PRT_LENGTH + 1; i++) {
246 l_arr[i] = index_of[l_arr[i]];
247 if (l_arr[i] != FF) {
248 l_val = i;
249 }
250 }
251
252 //Error Root Finder
253 memcpy(&reg[1], &l_arr[1], PRT_LENGTH * sizeof(reg[0]));
254 corrected_bits = 0;
255 for (i = 1, k = IPRIM - 1; i <= NN; i++, k = MODFF(k + IPRIM)) {
256 q = 1;
257 for (j = l_val; j > 0; j--)
258 if (reg[j] != FF) {
259 reg[j] = MODFF(reg[j] + j);
260 q ^= alpha_to[reg[j]];
261 }
262
263 if (q != 0) {
264 continue;
265 }
266 root[corrected_bits] = i;
267 loc[corrected_bits] = k;
268 if (++corrected_bits == l_val) {
269 break;
270 }
271 }
272
273 if (l_val != corrected_bits) {
274 return -1; //this fixes an error in the libre space code. It was returning 0 which it would also return if it didn't need to correct!
275 }
276
277 //Error Evaluator
278 eval_val = l_val - 1;
279 for (i = 0; i <= eval_val; i++) {
280 tmp = 0;
281 for (j = i; j >= 0; j--) {
282 if ((s[i - j] != FF) && (l_arr[j] != FF)) {
283 tmp ^= alpha_to[MODFF(s[i - j] + l_arr[j])];
284 }
285 }
286 evaluator[i] = index_of[tmp];
287 }
288
289 //Error Value Calculator
290 for (j = corrected_bits - 1; j >= 0; j--) {
291 num1 = 0;
292 for (i = eval_val; i >= 0; i--) {
293 if (evaluator[i] != FF) {
294 num1 ^= alpha_to[MODFF(evaluator[i] + i * root[j])];
295 }
296 }
297 num2 = alpha_to[MODFF(root[j] * (FCR_RS - 1) + NN)];
298 den = 0;
299
300 for (i = min(l_val, PRT_LENGTH - 1) & ~1; i >= 0; i -= 2) {
301 if (l_arr[i + 1] != FF) {
302 den ^= alpha_to[MODFF(l_arr[i + 1] + i * root[j])];
303 }
304 }
305
306 if (num1 != 0 && loc[j] >= pad) {
307 data[loc[j] - pad] ^= alpha_to[MODFF(index_of[num1] + index_of[num2] + NN -
308 index_of[den])];
309 }
310 }
311
312 return corrected_bits;
313}

790static void
791calculate_punctured_output(uint8_t *out, const uint8_t *data, const size_t len,
792 const uint8_t *puncturing_pattern_c1,
793 const uint8_t *puncturing_pattern_c2,
794 const uint8_t puncturing_pattern_length)
795{
796 uint8_t shift_reg_7[CCSDS_CONV_CONSTRAINT_LENGTH_BITS] = {0, 0, 0, 0, 0, 0, 0};
797 uint8_t bit = 0;
798 size_t processed = 0;
799 uint8_t input_bit = 0;
800 uint8_t out_bit = 0;
801 uint8_t out_g1 = 0;
802 uint8_t out_g2 = 0;
803 uint8_t puncturing_pattern_index = 0;
804
805 /* Clean output buffer from possible garbage value */
806 *out = 0;
807
808 while (processed < len * 8) {
809 /* Read each bit */
810 bit = read_bit(*data, input_bit++);
811 /* bit shift virtual register */
812 bit_shift(shift_reg_7, bit);
813 /* Calculate output for G1 and G2 */
814 out_g1 = calculate_generator_output(shift_reg_7, CCSDS_CONV_G1);
815 out_g2 = calculate_generator_output(shift_reg_7, CCSDS_CONV_G2);
816
817 /* Write convolution output to out */
818 *out |= (((out_g1 & puncturing_pattern_c1[puncturing_pattern_index])
819 << (7 - out_bit)));
820 out_bit += puncturing_pattern_c1[puncturing_pattern_index];
821 *out |= (((out_g2 & puncturing_pattern_c2[puncturing_pattern_index])
822 << (7 - out_bit)));
823 out_bit += puncturing_pattern_c2[puncturing_pattern_index];
824
825 /* Move pancture index to next value and reset if needed */
826 puncturing_pattern_index = (puncturing_pattern_index + 1) %
827 puncturing_pattern_length;
828
829 /* Increase processed bits */
830 processed++;
831 /* Check if processed 8 bits */
832 if (out_bit == 8) {
833 out++;
834 *out = 0;
835 out_bit = 0;
836 }
837 /* Check if out byte full */
838 if (input_bit == 8) {
839 data++;
840 input_bit = 0;
841 }
842 }
