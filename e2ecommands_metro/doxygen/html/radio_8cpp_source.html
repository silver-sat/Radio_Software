152/* beacon mode is entered by putting the AX5043 in Wire mode and setting the modulation for ASK.
153 * When in WIRE mode the data rate can instead be thought of as sampling rate.
154 * So, if you have a data rate of 100 bps, then the smallest time sample is 10 mSeconds.
155 * Since you're in wire mode, you're clocking out a 1 if data is high, every
156 * 1/datarate seconds.
157 *
158 * To enter wire mode write 0x84 to PINFUNCDATA (there's a library call for
159 * that).
160 *
161 * THIS IS IMPORTANT: The PA will go kablooey if you turn on the RF signal
162 * before the power is applied. I had three (at ~$20 each) PA's blow up this
163 * way. Make sure the output switch is set to the load (TX/~RX high, ~TX/RX
164 * low), and that a load is attached. You must also switch the TX path to
165 * single ended. Only then can you safely turn on the RF signal from the
166 * AX5043.
167 */
168void Radio::beaconMode(ax_config &config, ax_modulation &mod)
169{
170 ax_off(&config);
171 ax_init(&config); // do an init first
172 ax_default_params(&config, &mod); // load the RF parameters
173 digitalWrite(_pin_AX5043_DATA, LOW);
174
175 _func = 0x84; // set for wire mode:
176 //0x84 => 1000 0100 => PUDATA=1 (DATA weak Pullup enable), PFDATA=4 => DATA input/output modem data
177 ax_set_pinfunc_data(&config, _func); // remember to set this back when done!
178
179 // set the RF switch to transmit
180 digitalWrite(_pin_TX_RX, HIGH);
181 digitalWrite(_pin_RX_TX, LOW);
182
183 debug_printf("config variable values: \r\n");
184 debug_printf("tcxo frequency: %u \r\n", uint(config.f_xtal));
185 debug_printf("synthesizer A frequency: %u \r\n", uint(config.synthesiser.A.frequency));
186 debug_printf("synthesizer B frequency: %u \r\n", uint(config.synthesiser.B.frequency));
187 debug_printf("status: %x \r\n", ax_hw_status());
188 ax_SET_SYNTH_A(&config); //make sure we're using SYNTH A
189 ax_tx_on(&config, &ask_modulation);
190}

225/* cW mode is entered by putting the AX5043 in Wire mode and setting the modulation for ASK.
226 * When in WIRE mode the data rate can instead be thought of as sampling rate.
227 * So, if you have a data rate of 100 bps, then the smallest time sample is 10 mSeconds.
228 * Since you're in wire mode, you're clocking out a 1 if data is high, every
229 * 1/datarate seconds.
230 *
231 * To enter wire mode write 0x84 to PINFUNCDATA (there's a library call for
232 * that).
233 *
234 * THIS IS IMPORTANT: The PA will go kablooey if you turn on the RF signal
235 * before the power is applied. I had three (at ~$20 each) PA's blow up this
236 * way. Make sure the output switch is set to the load (TX/~RX high, ~TX/RX
237 * low), and that a load is attached. You must also switch the TX path to
238 * single ended. Only then can you safely turn on the RF signal from the
239 * AX5043.
240 */
241void Radio::cwMode(ax_config &config, ax_modulation &mod, int duration, ExternalWatchdog &watchdog)
242{
243 ax_init(&config); // do an init first
244 // modify the power to match what's in the modulation structure...make sure the modulation type matches
245 // this keeps beacon at full power
246 ask_modulation.power = mod.power;
247
248 debug_printf("ask power: %f \r\n", ask_modulation.power); //check to make sure it was modified...but maybe it wasn't?
249
250 ax_default_params(&config, &ask_modulation); // load the RF parameters
251
252 pinfunc_t _func = 0x84; // set for wire mode
253 ax_set_pinfunc_data(&config, _func); // remember to set this back when done!
254
255 // set the RF switch to transmit
256 digitalWrite(_pin_TX_RX, HIGH);
257 digitalWrite(_pin_RX_TX, LOW);
258 digitalWrite(_pin_AX5043_DATA, HIGH);
259
260 ax_tx_on(&config, &ask_modulation); // turn on the transmitter
261
262 // start transmitting
263 debug_printf("output CW for %u seconds \r\n", duration);
264 digitalWrite(_pin_PAENABLE, HIGH);
265 // delay(PAdelay); //let the pa bias stabilize
266 digitalWrite(_pin_TX_LED, HIGH);
267 digitalWrite(_pin_AX5043_DATA, HIGH);
268 int duration_timer_start = millis();
269 while (millis() - duration_timer_start < duration * 1000)
270 {
271 watchdog.trigger();
272 }
273
274 // stop transmitting
275 digitalWrite(_pin_AX5043_DATA, LOW);
276 digitalWrite(_pin_TX_RX, LOW); // put the switch back to receive
277 digitalWrite(_pin_RX_TX, HIGH);
278 digitalWrite(_pin_PAENABLE, LOW); // turn off the PA
279 digitalWrite(_pin_TX_LED, LOW);
280 debug_printf("done \r\n");
281
282 // drop out of wire mode
283 _func = 2;
284 ax_set_pinfunc_data(&config, _func);
285
286 // now put it back the way you found it.
287 ax_init(&config); // do a reset
288 ax_default_params(&config, &mod); // ax_modes.c for RF parameters
289 debug_printf("default params loaded \r\n");
290 // Serial.println("default params loaded \r\n");
291 ax_rx_on(&config, &mod);
292 debug_printf("receiver on \r\n");
293 // Serial.println("receiver on \r\n");
294}
